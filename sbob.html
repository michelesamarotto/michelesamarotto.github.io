*marble.pde: 

int c;
int w = 32;
int h = 32;
int file_count = 0;
float freq = 8.0;
float amp = 1.0;
float x_inc = 0.01;
float y_inc = 0.01;
float pixel_size = 0.125;
byte[] bytes;
String out_dir = "frames/frame";

void setup() {
  bytes = new byte[w * h];  
  size(w, h, P3D);
  background(0);
}

void draw() {
  float x_off = 0.0;

  loadPixels();

  for (int x = 0; x < width; x++) {
    float y_off = 0.0;

    for (int y = 0; y < height; y++) {

      c = stripe_turbulence(x, x_off, y_off, freq, pixel_size, amp);

      pixels[x + y * width] = color(c);
      bytes[x + y * width] = byte(c);

      y_off += y_inc;
    }

    x_off += x_inc;
  }

  updatePixels();
  saveBytes(out_dir + file_count, bytes);

  amp += 1.0;
  println(file_count);
  file_count += 1;
}

float turbulence(float x, float y, float pixel_size) {
  float t = 0;
  float scale = 1;

  while (scale > pixel_size) {
    t += abs(noise(x / scale, y / scale) * scale);
    scale /= 2;
  }

  return t;
}

int stripe_turbulence(int x, float x_off, float y_off, float freq, float pixel_size, float amp) {
  float t;

  t = (1 + cos((PI * (x + turbulence(x_off, y_off, pixel_size) * amp)) /freq)) / 2;   

  return int((1 - t) * 255);
}
